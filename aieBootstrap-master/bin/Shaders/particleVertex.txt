// VERTEX SHADER
#version 410 

layout(location=0) in vec4 Position; 
layout(location=1) in vec4 Colour;
layout(location=2) in vec2 UV;
 
out vec2 uv;
out vec4 colour; 

uniform mat4 projectionViewWorldMatrix;

void main() 
{
	uv = UV;
	colour = Colour; 
	gl_Position = projectionViewWorldMatrix * Position;
}
